17  import java.io.File;
18  
19  import org.apache.maven.plugins.annotations.LifecyclePhase;
20  import org.apache.maven.plugins.annotations.Mojo;
21  import org.apache.maven.plugins.annotations.Parameter;
22  import org.apache.maven.project.MavenProject;
23  
24  /**
25   * This mojo helps adding a filtered source folder in one go. This is typically useful if you want to use properties
26   * coming from the POM inside parts of your test source code that requires real constants, like annotations for example.
27   */
28  @Mojo( name = "filter-test-sources", defaultPhase = LifecyclePhase.GENERATE_TEST_SOURCES, threadSafe = true )
29  public class FilterTestSourcesMojo
30      extends AbstractFilterSourcesMojo
31  {
32      /**
33       * Source directory that will be first filtered and then added as a classical source folder.
34       */
35      @Parameter( defaultValue = "${basedir}/src/test/java-templates" )
36      private File testSourceDirectory;
37  
38      /**
39       * Output folder where filtered test sources will land.
40       */
41      @Parameter( defaultValue = "${project.build.directory}/generated-test-sources/java-templates" )
42      private File testOutputDirectory;
43  
44      @Override
45      protected File getSourceDirectory()
46      {
47          return testSourceDirectory;
48      }
49  
50      @Override
51      protected File getOutputDirectory()
52      {
53          return testOutputDirectory;
54      }
55  
56      @Override
57      protected void addSourceFolderToProject( MavenProject mavenProject )
58      {
59          mavenProject.addTestCompileSourceRoot( getOutputDirectory().getAbsolutePath() );
60      }
61  }
